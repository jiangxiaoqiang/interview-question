\begin{table}[htbp]
	\caption{Java类文件结构}
	\label{table:mapo2}
	\begin{center}
		\begin{tabular}{cp{5cm}c}
			\hline
			\multirow{1}{*}{占用大小}

			4bit &  magic：魔数，用于标识文件类型，对于java来说是0xCAFEBABE  &  1 \\

			2bit &  minor\_version：次版本号 & 1 \\
			\hline
			2bit &  major\_version：主版本号 & 1 \\
			\hline							
		\end{tabular}	
	\end{center}
\end{table}
